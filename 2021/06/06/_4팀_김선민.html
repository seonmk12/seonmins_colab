import random
r= random.randint(1,100)
p = int(input('예상숫자:'))
while r != p:
    if r > p:
        print('UP')
        p = int(input('예상숫자:'))
    elif r < p:
        print('DOWN')
        p = int(input('예상숫자:'))
    
print(f'정답은 {r}이 었습니다.')
